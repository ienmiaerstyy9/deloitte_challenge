# 3: Do transformation if a condition is met, else do a different transformation. Store in new column

def compute_apply_step3(row):
    if row['sensor_4'] < median_sensor_4:
        return row['sensor_2'] * row['sensor_3']
    else:
        calendar_month_mean = df_filtered[df_filtered['date_time'].dt.month == row['date_time'].month]['sensor_4'].mean()
        return row['sensor_2'] * row['sensor_4'] - calendar_month_mean
        
median_sensor_4 = df_filtered['sensor_4'].median()
df_filtered['apply_step3'] = df_filtered.apply(compute_apply_step3, axis=1)

# 5: Create a function for splits and make it generic to arbitrary splits.
def create_split_column(data, column_name, splits):
    """
    Normalizes and splits a column at user supplied split points
    """
    normalized_col = (data[column_name] - data[column_name].min()) / (data[column_name].max() - data[column_name].min())
    return pd.cut(normalized_col, bins=splits, labels=False, include_lowest=True)

split_points = [0, 0.2, 0.5, 0.8, 0.85, 1.0]

df_full['Truth'] = create_split_column(df_full, 'sensor_1', split_points)
df_full['Predicted'] = create_split_column(df_full, 'sensor_2', split_points)
